11 — Formal Grammar (EBNF)

Normative. Lexer-resolved: comments, Roman numerals (→ INT), string interpolation (STRING → interleaved text/expression parts), line-continuation (01-lexical §7). NL = newline token. Target: Lark (lalr preferred; earley acceptable for the reference implementation).


(* ===== file structure ===== *)
file            = [salutation] header {NL} {top_stmt} ;
salutation      = "AVE" "OMNISSIAH" "." NL ;
header          = ("CANTICUM" | "FABRICA" | "LITANIA") STRING "." NL ;

top_stmt        = import_stmt | schema_def | rite_def | stmt | infra_stmt ;
(* mode restrictions (P-XVI) are enforced post-parse:
   LITANIA: import_stmt, schema_def, rite_def, declare only
   CANTICUM: no infra_stmt ; FABRICA: everything *)

(* ===== imports ===== *)
import_stmt     = "INVOCO" STRING ["UT" IDENT] NL
                | "INVOCO" IDENT {"," IDENT} "EX" STRING NL ;

(* ===== declarations ===== *)
schema_def      = "SCHEMA" IDENT NL {field_decl} "FINIS" "SCHEMA" NL ;
field_decl      = IDENT ":" type ["=" expr] NL ;

rite_def        = ["SANCTUM"] "RITUS" IDENT "(" [params] ")" "->" type NL
                  block "FINIS" "RITUS" NL ;
params          = param {"," param} ;
param           = IDENT ":" type ;

declare         = ("DECLARO" | "SANCTUM") IDENT ":" type "=" expr NL ;

(* ===== statements ===== *)
stmt            = declare | assign | if_stmt | while_stmt | for_stmt
                | try_stmt | print_stmt | return_stmt | raise_stmt
                | expr_stmt | "RUMPO" NL | "PERGO" NL ;

assign          = lvalue "FIAT" expr NL ;
lvalue          = IDENT {"[" expr "]" | "." IDENT} ;

if_stmt         = "SI" expr "TUNC" NL block
                  {"ALITER" "SI" expr "TUNC" NL block}
                  ["ALITER" NL block]
                  "FINIS" "SI" NL ;
while_stmt      = "DUM" expr "AGE" NL block "FINIS" "DUM" NL ;
for_stmt        = "PRO" "OMNI" IDENT "IN" expr "AGE" NL block "FINIS" "PRO" NL ;
block           = {stmt | rite_def_local} ;
rite_def_local  = rite_def ;   (* legal in rite bodies only, P-XIII elsewhere *)

try_stmt        = "TEMPTA" NL block
                  ["SI" "HERESIS" "(" IDENT ")" NL block]
                  ["DENIQUE" NL block]
                  "FINIS" "TEMPTA" NL ;
(* at least one of catch/finally required: P-XIV if both absent *)

print_stmt      = ("VOCIFERO" | "SUSURRO") expr NL ;
return_stmt     = "REDDO" [expr] NL ;
raise_stmt      = "PROCLAMO" "HERESIM" expr ["GENERIS" STRING] NL ;
expr_stmt       = postfix_expr NL ;   (* call statements *)

(* ===== types ===== *)
type            = "NUMERUS" | "FRACTIO" | "SCRIPTUM" | "VERITAS" | "NIHIL"
                | "RITUS" | "HERESIS"
                | "ORDO" "[" type "]"
                | "TABULA" "[" type "," type "]"
                | IDENT ;                    (* schema name; P-XV if unknown *)

(* ===== expressions ===== *)
expr            = or_expr ;
or_expr         = and_expr {"VEL" and_expr} ;
and_expr        = eq_expr {"ET" eq_expr} ;
eq_expr         = cmp_expr [("==" | "!=") cmp_expr] ;
cmp_expr        = add_expr [("<" | "<=" | ">" | ">=") add_expr] ;
(* single comparison only: chaining is P-IX *)
add_expr        = mul_expr {("+" | "-") mul_expr} ;
mul_expr        = unary {("*" | "/" | "%") unary} ;
unary           = ["-" | "NON"] postfix_expr ;
postfix_expr    = primary {"(" [args] ")" | "[" expr "]" | "." IDENT} ;
args            = expr {"," expr} [","] ;

primary         = INT | FLOAT | string | "VERUM" | "FALSUM" | "NIHIL"
                | IDENT | "(" expr ")"
                | list_lit | map_lit | schema_lit | rite_expr ;

string          = STRING_START {STRING_TEXT | "{" expr "}"} STRING_END ;
list_lit        = "[" [expr {"," expr} [","]] "]" ;
map_lit         = "{" [map_entry {"," map_entry} [","]] "}" ;
map_entry       = expr ":" expr ;
schema_lit      = IDENT "{" [field_init {"," field_init} [","]] "}" ;
field_init      = IDENT ":" expr ;
rite_expr       = "RITUS" "(" [params] ")" "->" type NL block "FINIS" "RITUS" ;

(* ===== infrastructure (FABRICA only) ===== *)
infra_stmt      = covenant | seat | resource | scrutiny | demand | proclaim ;

covenant        = "FOEDUS" STRING NL attr_block "FINIS" "FOEDUS" NL ;
seat            = "SEDES" STRING NL attr_block "FINIS" "SEDES" NL ;
resource        = "EXSTRUATUR" IDENT string_or_interp NL attr_block
                  "FINIS" "EXSTRUATUR" NL ;
scrutiny        = "SCRUTOR" IDENT STRING NL attr_block "FINIS" "SCRUTOR" NL ;
demand          = "POSTULO" IDENT ":" type ["=" expr] ["ARCANUM"] NL ;
proclaim        = "PROFITEOR" IDENT "=" expr ["ARCANUM"] NL ;

attr_block      = {attr} ;
attr            = IDENT ":" (expr NL | NL attr_block "FINIS" NL) ;
string_or_interp = string ;

Parse ambiguity notes (normative resolutions)

  1. schema_lit vs map_lit vs block-open: IDENT "{" at expression position is a schema literal; bare "{" is a map. In statement position, an IDENT "{" sequence cannot occur (calls need (), so no conflict with attr blocks.
  2. SI HERESIS inside try_stmt is distinguished from an if_stmt by the HERESIS ( lookahead; LALR handles it because HERESIS is a keyword.
  3. SANCTUM leads either declare or rite_def — decided by the following token (IDENT : vs RITUS).
  4. schema_lit requires the IDENT be a known schema at post-parse resolution; unknown → P-XV (not a parse error, better message).
  5. Keywords ARCANUM/GENERIS appear only in their productions; they are reserved globally (add to 01-lexical §4 master list — done in rev 2).
