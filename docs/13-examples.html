3. Schemas, closures, list rites


AVE OMNISSIAH.
CANTICUM "census_servitorum".

INVOCO "ordo_opera" UT oo

SCHEMA servitor
    nomen : SCRIPTUM
    gradus : NUMERUS = I
FINIS SCHEMA

RITUS promotio_facta (minimus_gradus : NUMERUS) -> RITUS
    REDDO RITUS (s : servitor) -> VERITAS
        REDDO s.gradus >= minimus_gradus     ++ closure over minimus_gradus
    FINIS RITUS
FINIS RITUS

DECLARO cohors : ORDO[servitor] = [
    servitor { nomen: "Kane", gradus: VII },
    servitor { nomen: "Vex" },
    servitor { nomen: "Ohm", gradus: IV },
]

DECLARO digni : ORDO[servitor] = oo.excerne(cohors, promotio_facta(IV))
PRO OMNI s IN digni AGE
    VOCIFERO "{s.nomen} — rank {romanum(s.gradus)} — worthy."
FINIS PRO

4. Heresy handling with re-raise


AVE OMNISSIAH.
CANTICUM "auspex_lectio".

INVOCO "codex_json"
INVOCO "archivum"

RITUS lege_configurationem (via : SCRIPTUM) -> TABULA[SCRIPTUM, SCRIPTUM]
    TEMPTA
        REDDO codex_json.lege(archivum.lege(via))
    SI HERESIS (delictum)
        SI delictum.genus == "archivum_ignotum" TUNC
            SUSURRO "No config scroll at {via}; doctrine defaults apply."
            REDDO {}
        FINIS SI
        PROCLAMO HERESIM delictum        ++ not ours to absolve — re-raise
    FINIS TEMPTA
FINIS RITUS

VOCIFERO scriptum(lege_configurationem("configuratio.json"))

6. A publishable litania with tests

litania.toml:


[litania]

descriptio = "Tithe calculations, Administratum-grade"
licentia   = "MIT"
gothica    = ">=0.2"

fons/tributum.vg:


AVE OMNISSIAH.
LITANIA "tributum".

INVOCO fractio_integra EX "mathematica"

SANCTUM pars_decima : FRACTIO = 0.1

RITUS summa_tributi (census : NUMERUS) -> NUMERUS
    SI census < N TUNC
        PROCLAMO HERESIM "Negative census defies the Administratum" GENERIS "census_falsus"
    FINIS SI
    REDDO census + fractio_integra(fractio(census) * pars_decima)
FINIS RITUS

probationes/tributum_proba.vg:


AVE OMNISSIAH.
LITANIA "tributum_proba".

INVOCO adfirma_aequalia, adfirma_heresim EX "probatio"
INVOCO "fons/tributum"

RITUS proba_summa () -> NIHIL
    adfirma_aequalia(tributum.summa_tributi(C), CX, "tithe of C is CX")
FINIS RITUS

RITUS proba_census_falsus () -> NIHIL
    adfirma_heresim("census_falsus", RITUS () -> NIHIL
        DECLARO _x : NUMERUS = tributum.summa_tributi(-I)
    FINIS RITUS)
FINIS RITUS

Ship it: gothica proba && git tag v0.1.0 && gothica offero
